# Question 1: Does altitude affect blood oxygen levels?

# The most basic test to do here is a Mann Whitney U test.
# This is a non-parametic test.
# Null hypothesis: It is equally likely that a randomly selected value from one population will be less than or greater than a randomly selected value from a second population.

# The p-value assumes a normal distribution, though the statistic does not.
# This data is technically repeat measurements of one person (me), so there is correlation.
stat, p = mannwhitneyu(
    treatment["k_bloodoxy"].tolist(),
    control["k_bloodoxy"].tolist(),
    alternative="two-sided",
)
print(stat)
print(p)

# Since the sample sizes are so small we can calculate test statistics by hand
# This version doesn't really account for ties
# Time complexity: 2 * n * m (n = # in group 1, m = # in group 2)


def exact_mann_whitney(treatment, control, col):
    U1 = 0
    for num in treatment[col].tolist():
        bigger = [x for x in control[col].tolist() if x > num]
        U1 += len(bigger)
    print("U1", U1)

    U2 = 0
    for num in control[col].tolist():
        bigger = [x for x in treatment[col].tolist() if x > num]
        U2 += len(bigger)
    print("U2", U2)
    U = min(U1, U2)
    return U


U = exact_mann_whitney(treatment, control, "k_bloodoxy")
print("U statistic", U)

# Another way to calculate the Mann Whitney U statistic with lower time complexity implemented in scipy

x = np.asarray(treatment["k_bloodoxy"].tolist())
y = np.asarray(control["k_bloodoxy"].tolist())

arr = np.concatenate((x, y))
sorter = np.argsort(arr, kind="quicksort")  # returns indices that would sort array
inv = np.zeros(sorter.size, dtype=np.intp)  # 0 array
inv[sorter] = np.arange(sorter.size, dtype=np.intp)
arr = arr[sorter]  # Now the array is sorted
obs = np.r_[
    True, arr[1:] != arr[:-1]
]  # Array where 'True' indicates first of a unique value
dense = obs.cumsum()[inv]  # rank each value for 1 - num_unique_values
count = np.r_[
    np.nonzero(obs)[0], len(obs)
]  # np.nonzero(obs)[0] is where the non-zero values are located, indicating the next unique value
ranked = 0.5 * (
    count[dense] + count[dense - 1] + 1
)  # For each set of non-unique values, get the middle value
# Now they are in the initial order, but with their rank
rankx = ranked[0 : len(x)]  # get the x-ranks
u1 = (
    len(x) * len(y) + (len(x) * (len(x) + 1)) / 2.0 - np.sum(rankx, axis=0)
)  # calc U for x
u2 = len(x) * len(y) - u1  # remainder is U for y
print(u1)
print(u2)
